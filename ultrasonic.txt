#include <WiFi.h>

char ssid[] = "AirfibreSujay";
char pass[] = "Octane0027";

#define BLYNK_PRINT Serial

Servo myServo;
const int servoPin = 18;
const int OPEN_ANGLE = 90;
const int CLOSE_ANGLE = 0;

#define TRIG_PIN 32
#define ECHO_PIN 34
long duration;
float distanceCM;

BlynkTimer timer;
WidgetRTC rtc;

bool notified2pm = false;
bool notified520pm = false;

bool foodWasPresent = true;  // Flag to detect food presence changes

void moveServo(int targetAngle) {
  int currentAngle = myServo.read();
  while (currentAngle != targetAngle) {
    currentAngle += (targetAngle > currentAngle) ? 1 : -1;
    myServo.write(currentAngle);
    delay(15);
  }
}

BLYNK_WRITE(V0) {
  int pinValue = param.asInt();
  if (pinValue == 1) {
    Serial.println("Manual feed: Opening feeder...");
    moveServo(OPEN_ANGLE);
    delay(1000);
    moveServo(CLOSE_ANGLE);
  }
}

BLYNK_WRITE(V1) {
  int pinValue = param.asInt();
  if (pinValue == 1) {
    Serial.println("Auto feed triggered by automation.");
    moveServo(OPEN_ANGLE);
    delay(1000);
    moveServo(CLOSE_ANGLE);
  }
}

void checkSchedule() {
  int hr = hour();
  int min = minute();

  if (hr == 14 && min == 0 && !notified2pm) {
    Blynk.logEvent("feeding_reminder", "It's 2 PM! Time to feed your pet 🐾");
    Serial.println("Notification sent: 2 PM feeding reminder");
    notified2pm = true;
  }

  if (hr == 17 && min == 20 && !notified520pm) {
    Blynk.logEvent("feeding_reminder", "It's 5:20 PM! Time to feed your pet 🐶🍽");
    Serial.println("Notification sent: 5:20 PM feeding reminder");
    notified520pm = true;
  }

  if (hr == 0 && min == 1) {
    notified2pm = false;
    notified520pm = false;
    Serial.println("Daily reset: Notification flags reset.");
  }
}

void checkFoodLevel() {
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);

  duration = pulseIn(ECHO_PIN, HIGH, 30000);
  distanceCM = duration * 0.034 / 2;

  Serial.print("Distance: ");
  Serial.print(distanceCM);
  Serial.println(" cm");

  Blynk.virtualWrite(V3, distanceCM);

  if (distanceCM >= 5.0) {
    if (foodWasPresent) {
      Blynk.logEvent("food_low", "⚠ Feeder is empty! Please refill.");
      Serial.println("⚠ Notification sent: Food is low!");
      foodWasPresent = false;
    }
  } else {
    if (!foodWasPresent) {
      Serial.println("✅ Food detected again. Resetting flag.");
      foodWasPresent = true;
    }
  }
}

void reconnectWiFi() {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("Wi-Fi disconnected. Reconnecting...");
    WiFi.begin(ssid, pass);
    int retries = 0;
    while (WiFi.status() != WL_CONNECTED && retries < 20) {
      delay(500);
      Serial.print(".");
      retries++;
    }
    Serial.println(WiFi.status() == WL_CONNECTED ? "\nWi-Fi reconnected!" : "\nWi-Fi reconnection failed.");
  }
}

void setup() {
  Serial.begin(115200);
  delay(1000);

  myServo.attach(servoPin);
  myServo.write(CLOSE_ANGLE);

  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);

  WiFi.begin(ssid, pass);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWi-Fi connected.");

  Blynk.config(BLYNK_AUTH_TOKEN);
  Blynk.connect();

  rtc.begin();

  timer.setInterval(60000L, checkSchedule);
  timer.setInterval(10000L, checkFoodLevel);
}

void loop() {
  Blynk.run();
  timer.run();
  reconnectWiFi();
}
